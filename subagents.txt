# Key Commands
commands:
  - name: /loop
    description: Unified autonomous coding loop with interactive, AFK, and single modes
    example: /loop "Add authentication" --afk --max 30

  - name: /remember
    description: Save facts to persistent memory
    example: /remember API uses JWT tokens

  - name: /recall
    description: Search accumulated knowledge
    example: /recall database schema

  - name: /pv-mesh
    description: Parallel multi-AI verification (Claude + Codex + Gemini)
    example: /pv-mesh "Is this architecture correct?"

  - name: /evolve
    description: Run self-improvement cycle to detect patterns and clean bloat
    example: /evolve

  - name: /research-swarm
    description: Launch parallel research agents on multiple topics
    example: /research-swarm "React vs Vue vs Svelte"

# Installation
installation:
  type: git-clone
  steps:
    - command: git clone https://github.com/brolag/neural-claude-code-plugin ~/Sites/neural-claude-code-plugin
      description: Clone the repository
    - command: cd ~/Sites/neural-claude-code-plugin && ./scripts/setup-hooks.sh
      description: Setup Claude Code hooks
    - command: export CLAUDE_PLUGIN_ROOT="$HOME/Sites/neural-claude-code-plugin"
      description: Add to shell profile (~/.zshrc or ~/.bashrc)
